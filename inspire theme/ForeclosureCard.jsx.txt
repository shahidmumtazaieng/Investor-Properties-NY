import React from 'react';
import { Link } from 'react-router-dom';
import { 
  MapPinIcon, 
  CalendarIcon, 
  CurrencyDollarIcon,
  Building2Icon,
  ArrowRightIcon,
  AlertTriangleIcon
} from 'lucide-react';
import Card from '../ui/Card';
import Button from '../ui/Button';

const ForeclosureCard = ({ foreclosure, animationDelay = 0 }) => {
  const formatDate = (dateString) => {
    const date = new Date(dateString);
    return date.toLocaleDateString('en-US', { 
      year: 'numeric', 
      month: 'long', 
      day: 'numeric' 
    });
  };

  const daysUntilAuction = Math.ceil((new Date(foreclosure.auctionDate) - new Date()) / (1000 * 60 * 60 * 24));

  return (
    <Card 
      variant="elevated" 
      className="overflow-hidden group animate-slide-up border-l-4 border-l-accent-yellow"
      style={{ animationDelay: `${animationDelay}s` }}
    >
      {/* Header */}
      <div className="bg-gradient-to-r from-accent-yellow/10 to-accent-yellow/5 p-4">
        <div className="flex items-center justify-between mb-2">
          <div className="flex items-center">
            <AlertTriangleIcon className="w-5 h-5 text-accent-yellow mr-2" />
            <span className="text-sm font-semibold text-accent-yellow uppercase tracking-wide">
              Foreclosure Auction
            </span>
          </div>
          <span className="text-xs text-neutral-600">
            {daysUntilAuction > 0 ? `${daysUntilAuction} days left` : 'Auction ended'}
          </span>
        </div>
      </div>

      {/* Property Details */}
      <div className="p-6">
        {/* Address */}
        <div className="mb-4">
          <h3 className="text-lg font-bold text-primary-blue mb-1 group-hover:text-accent-yellow transition-colors">
            {foreclosure.address}
          </h3>
          <div className="flex items-center text-neutral-600 text-sm">
            <MapPinIcon className="w-4 h-4 mr-1" />
            {foreclosure.county} County
          </div>
        </div>

        {/* Property Info */}
        <div className="space-y-3 mb-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center text-neutral-600 text-sm">
              <Building2Icon className="w-4 h-4 mr-2" />
              <span>Property Type:</span>
            </div>
            <span className="text-sm font-semibold text-neutral-800">{foreclosure.propertyType}</span>
          </div>
          
          <div className="flex items-center justify-between">
            <div className="flex items-center text-neutral-600 text-sm">
              <CalendarIcon className="w-4 h-4 mr-2" />
              <span>Auction Date:</span>
            </div>
            <span className="text-sm font-semibold text-neutral-800">
              {formatDate(foreclosure.auctionDate)}
            </span>
          </div>
          
          <div className="flex items-center justify-between">
            <div className="flex items-center text-neutral-600 text-sm">
              <CurrencyDollarIcon className="w-4 h-4 mr-2" />
              <span>Starting Bid:</span>
            </div>
            <span className="text-xl font-bold text-accent-yellow">
              {foreclosure.startingBid}
            </span>
          </div>
        </div>

        {/* Action Buttons */}
        <div className="space-y-3">
          <Link to={`/foreclosures/${foreclosure.id}`} className="block">
            <Button variant="secondary" className="w-full group">
              View Details
              <ArrowRightIcon className="w-4 h-4 ml-2 group-hover:translate-x-1 transition-transform" />
            </Button>
          </Link>
          
          <div className="flex gap-2">
            <Button variant="outline" className="flex-1 text-sm py-2">
              Get Alert
            </Button>
            <Button variant="ghost" className="flex-1 text-sm py-2 text-accent-yellow">
              Save
            </Button>
          </div>
        </div>

        {/* Disclaimer */}
        <div className="mt-4 pt-4 border-t border-neutral-200">
          <p className="text-xs text-neutral-500 text-center">
            Auction details subject to change. Verify with county records.
          </p>
        </div>
      </div>
    </Card>
  );
};

export default ForeclosureCard;
